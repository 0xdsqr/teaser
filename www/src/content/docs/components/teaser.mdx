The main container component that wraps all teaser content.

## Usage

```tsx
import { Teaser } from "@/components/ui/teaser";

export default function ComingSoon() {
  return <Teaser className="max-w-md">{/* Your content */}</Teaser>;
}
```

## Props

| Prop        | Type      | Default |
| ----------- | --------- | ------- |
| `children`  | ReactNode | -       |
| `className` | string    | -       |
